35class MapTile(object):
36 def __init__(
37 self,
38 x: int = None,
39 y: int = None,
40 z: int = None,
41 filespec: str = None,
42 tile: "MapTile" = None,
43 suffix="jpg",
44 ):
45 """This is a simple wrapper around mercantile.tile to associate a
46 filespec with the grid coordinates.
47
48 Args:
49 x (int): The X index for this tile
50 y (int): The Y index for this tile
51 z (int): The Z index for this tile if there is one
52 filespec (str): The location of this within the map tile cache
53 tile (MapTile): Make a copy of this object
54 suffix (str): The image suffix, jpg or png usually
55
56 Returns:
57 (MapTile): An instance of this object
58 """
59 if tile:
60 self.x = tile.x
61 self.y = tile.y
62 self.z = tile.z
63 else:
64 self.x = x
65 self.y = y
66 self.z = z
67 self.blob = None
68 self.filespec = None
69 if not filespec and self.z:
70 self.filespec = f"{self.z}/{self.y}/{self.x}.{suffix}"
71 elif filespec:
72 self.filespec = filespec
73 tmp = filespec.split("/")
74 self.z = tmp[0]
75 self.x = tmp[2]
76 self.y = tmp[1].replace("." + suffix, "")

78 def readImage(self, base: str = "./"):
79 """Read a map tile out of the disk based map tile cache.
80
81 Args:
82 base (str): The top level directory for the map tile cache
83 """
84 file = f"{base}/{self.filespec}"
85 logging.debug("Adding tile image: %s" % file)
86 if os.path.exists(file):
87 size = os.path.getsize(file)
88 file = open(file, "rb")
89 self.blob = file.read(size)

158 self,
159 dbname: str,
160 append: bool = False,
161 ):
162 """Create and sqlitedb in either mbtiles or Osman sqlitedb format.
163
164 Args:
165 dbname (str): The filespec of the sqlite output file
166 """
167 suffix = os.path.splitext(dbname)[1]
168
169 if os.path.exists(dbname) and append == False:
170 os.remove(dbname)
171
172 self.db = sqlite3.connect(dbname)
173 self.cursor = self.db.cursor()
174 self.cursor.execute("SELECT name FROM sqlite_master WHERE type='table' AND name='tiles'")
175 exists = self.cursor.fetchone()
176 if exists and append:
177 logging.info("Appending to database file %s" % dbname)
178 return
179
180 if suffix == ".mbtiles":
181 self.cursor.execute("CREATE TABLE tiles (zoom_level integer, tile_column integer, tile_row integer, tile_data blob)")
182 self.cursor.execute("CREATE INDEX tiles_idx on tiles (zoom_level, tile_column, tile_row)")
183 self.cursor.execute("CREATE TABLE metadata (name text, value text)")
184 # These get populated later
185 name = dbname
186 description = "Created by osm_fieldwork/basemapper.py"
187 self.cursor.execute("CREATE UNIQUE INDEX metadata_idx ON metadata (name)")
188 self.cursor.execute("INSERT INTO metadata (name, value) VALUES('version', '1.1')")
189 self.cursor.execute("INSERT INTO metadata (name, value) VALUES('type', 'baselayer')")
190 self.cursor.execute(f"INSERT INTO metadata (name, value) VALUES('name', '{name}')")
191 self.cursor.execute(f"INSERT INTO metadata (name, value) VALUES('description', '{description}')")
192 # self.cursor.execute(f"INSERT INTO metadata (name, value) VALUES('bounds', '{bounds}')")
193 self.cursor.execute("INSERT INTO metadata (name, value) VALUES('format', 'jpg')")
194 if "sqlite" in suffix:
195 # s is always 0
196 self.cursor.execute("CREATE TABLE tiles (x int, y int, z int, s int, image blob, PRIMARY KEY (x,y,z,s));")
197 self.cursor.execute("CREATE INDEX IND on tiles (x,y,z,s)")
198 # Info is simple "2|4" for example, it gets populated later

219 def writeTile(
220 self,
221 tile: MapTile,
222 ):
223 """Write a map tile into the sqlite database file.
224
225 Args:
226 tile (MapTile): The map tile to write to the file
227 """
228 if tile.blob is None:
229 logging.error(f"Map tile {tile.filespec} has no image data!")
230 # tile.dump()
231 return False
232
233 suffix = os.path.splitext(self.dbname)[1]
234
235 if "sqlite" in suffix:
236 # Osmand tops out at zoom level 16, so the real zoom level is inverse,
237 # and can go negative for really high zoom levels.
238 z = 17 - tile.z
239 self.db.execute(
240 "INSERT INTO tiles (x, y, z, s, image) VALUES (?, ?, ?, ?, ?)",
241 [tile.x, tile.y, z, 0, sqlite3.Binary(tile.blob)],
242 )
243
244 if suffix == ".mbtiles":
245 y = (1 << tile.z) - tile.y - 1
246 self.db.execute(
247 "INSERT INTO tiles (tile_row, tile_column, zoom_level, tile_data) VALUES (?, ?, ?, ?)",
248 [y, tile.x, tile.z, sqlite3.Binary(tile.blob)],
249 )
250
251 self.db.commit()
