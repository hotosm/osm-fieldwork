63 def CSVparser(
64 self,
65 filespec: str,
66 data: str = None,
67 ) -> list:
68 """Parse the CSV file from ODK Central and convert it to a data structure.
69
70 Args:
71 filespec (str): The file to parse.
72 data (str): Or the data to parse.
73
74 Returns:
75 (list): The list of features with tags
76 """
77 all_tags = list()
78 if not data:
79 f = open(filespec, newline="")
80 reader = csv.DictReader(f, delimiter=",")
81 else:
82 reader = csv.DictReader(data, delimiter=",")
83 for row in reader:
84 tags = dict()
85 # log.info(f"ROW: {row}")
86 for keyword, value in row.items():
87 if keyword is None or value is None:
88 continue
89 if len(value) == 0:
90 continue
91 base = basename(keyword).lower()
92 # There's many extraneous fields in the input file which we don't need.
93 if base is None or base in self.ignore or value is None:
94 continue
95 else:
96 # log.info(f"ITEM: {keyword} = {value}")
97 if base in self.typestypes:
98 if self.typestypes[base] == "select_multiple":
99 vals = self.convertMultiple(value)
100 if len(vals) > 0:
101 tags.update(vals)
102 continue
103 # When using geopoint warmup, once the display changes to the map
104
105 # location, there is not always a value if the accuracy is way
106 # off. In this case use the warmup value, which is where we are
107 # hopefully standing anyway.
108 if base == "latitude" and len(value) == 0:
109 if "warmup-Latitude" in row:
110 value = row["warmup-Latitude"]
111 if base == "longitude" and len(value) == 0:
112 value = row["warmup-Longitude"]
113 items = self.convertEntry(base, value)
114 # log.info(f"ROW: {base} {value}")
115 if len(items) > 0:
116 if base in self.savedsaved:
117 if str(value) == "nan" or len(value) == 0:
118 # log.debug(f"FIXME: {base} {value}")
119 val = self.savedsaved[base]
120 if val and len(value) == 0:
121 log.warning(f'Using last saved value for "{base}"! Now "{val}"')
122 value = val
123 else:
124 self.savedsaved[base] = value
125 log.debug(f'Updating last saved value for "{base}" with "{value}"')
126 # Handle nested dict in list
127 if isinstance(items, list):
128 items = items[0]
129 for k, v in items.items():
130 tags[k] = v
131 else:
132 tags[base] = value
133 # log.debug(f"\tFIXME1: {tags}")
134 all_tags.append(tags)
135 return all_tags

137 def JSONparser(
138 self,
139 filespec: str = None,
140 data: str = None,
141 ) -> list:
142 """Parse the JSON file from ODK Central and convert it to a data structure.
143 The input is either a filespec to open, or the data itself.
144
145 Args:
146 filespec (str): The JSON or GeoJson input file to convert
147 data (str): The data to convert
148
149 Returns:
150 (list): A list of all the features in the input file
151 """
152 log.debug(f"Parsing JSON file {filespec}")
153 total = list()
154 if not data:
155 file = open(filespec, "r")
156 infile = Path(filespec)
157 if infile.suffix == ".geojson":
158 reader = geojson.load(file)
159 elif infile.suffix == ".json":
160 reader = json.load(file)
161 else:
162 log.error("Need to specify a JSON or GeoJson file!")
163 return total
164 elif isinstance(data, str):
165 reader = geojson.loads(data)
166 elif isinstance(data, list):
167 reader = data
168
169 # JSON files from Central use value as the keyword, whereas
170 # GeoJSON uses features for the same thing.
171 if "value" in reader:
172 data = reader["value"]
173 elif "features" in reader:
174 data = reader["features"]
175 else:
176 data = reader
177 for row in data:
178 # log.debug(f"ROW: {row}\n")
179 tags = dict()
180 if "properties" in row:
181 row["properties"] # A GeoJson formatted file
182 else:
183 pass # A JOSM file from ODK Central
184
185 # flatten all the groups into a sodk2geojson.pyingle data structure
186 flattened = flatdict.FlatDict(row)
187 # log.debug(f"FLAT: {flattened}\n")
188 for k, v in flattened.items():
189 last = k.rfind(":") + 1
190 key = k[last:]
191 # a JSON file from ODK Central always uses coordinates as
192 # the keyword
193 if key is None or key in self.ignore or v is None:
194 continue
195 # log.debug(f"Processing tag {key} = {v}")
196 if key == "coordinates":
197 if isinstance(v, list):
198 tags["lat"] = v[1]
199 tags["lon"] = v[0]
200 # poi = Point(float(lon), float(lat))
201 # tags["geometry"] = poi
202 continue
203
204 if key in self.typestypes:
205 if self.typestypes[key] == "select_multiple":
206 # log.debug(f"Found key '{self.types[key]}'")
207 if v is None:
208 continue
209 vals = self.convertMultiple(v)
210 if len(vals) > 0:
211 tags.update(vals)
212 continue
213 items = self.convertEntry(key, v)
214 if items is None or len(items) == 0:
215 continue
216
217 if type(items) == str:
218 log.debug(f"string Item {items}")
219 elif type(items) == list:
220 # log.debug(f"list Item {items}")
221 tags.update(items[0])
222 elif type(items) == dict:
223 # log.debug(f"dict Item {items}")
224 tags.update(items)
225 # log.debug(f"TAGS: {tags}")
226 if len(tags) > 0:
227 total.append(tags)
228
229 # log.debug(f"Finished parsing JSON file {filespec}")
230 return total

232 def XMLparser(
233 self,
234 filespec: str,
235 data: str = None,
236 ) -> list:
237 """Import an ODK XML Instance file ito a data structure. The input is
238 either a filespec to the Instance file copied off your phone, or
239 the XML that has been read in elsewhere.
240
241 Args:
242 filespec (str): The filespec to the ODK XML Instance file
243 data (str): The XML data
244
245 Returns:
246 (list): All the entries in the OSM XML Instance file
247 """
248 row = dict()
249 if filespec:
250 logging.info("Processing instance file: %s" % filespec)
251 file = open(filespec, "rb")
252 # Instances are small, read the whole file
253 xml = file.read(os.path.getsize(filespec))
254 elif data:
255 xml = data
256 doc = xmltodict.parse(xml)
257
258 json.dumps(doc)
259 tags = dict()
260 data = doc["data"]
261 flattened = flatdict.FlatDict(data)
262 # total = list()
263 # log.debug(f"FLAT: {flattened}")
264 pat = re.compile("[0-9.]* [0-9.-]* [0-9.]* [0-9.]*")
265 for key, value in flattened.items():
266 if key[0] == "@" or value is None:
267 continue
268 # Get the last element deliminated by a dash
269 # for CSV & JSON, or a colon for ODK XML.
270 base = basename(key)
271 log.debug(f"FLAT: {base} = {value}")
272 if base in self.ignore:
273 continue
274 if re.search(pat, value):
275 gps = value.split(" ")
276 row["lat"] = gps[0]
277 row["lon"] = gps[1]
278 continue
279
280 if base in self.typestypes:
281 if self.typestypes[base] == "select_multiple":
282 # log.debug(f"Found key '{self.types[base]}'")
283 vals = self.convertMultiple(value)
284 if len(vals) > 0:
285 tags.update(vals)
286 continue
287 else:
288 item = self.convertEntry(base, value)
289 if item is None or len(item) == 0:
290 continue
291 if len(tags) == 0:
292 tags = item[0]
293 else:
294 if type(item) == list:
295 # log.debug(f"list Item {item}")
296 tags.update(item[0])
297 elif type(item) == dict:
298 # log.debug(f"dict Item {item}")
299 tags.update(item)
300 row.update(tags)
301 return [row]
