59def downloadThread(project_id: int, xforms: list, odk_credentials: dict, filters: dict = None):
60 """Download a list of submissions from ODK Central.
61
62 Args:
63 project_id (int): The ID of the project on ODK Central
64 xforms (list): A list of the XForms to down the submissions from
65 odk_credentials (dict): The authentication credentials for ODK Collect
66
67 Returns:
68 (list): The submissions in JSON format
69 """
70 timer = Timer(text="downloadThread() took {seconds:.0f}s")
71 timer.start()
72 data = list()
73 # log.debug(f"downloadThread() called! {len(xforms)} xforms")
74 for task in xforms:
75 form = OdkForm(odk_credentials["url"], odk_credentials["user"], odk_credentials["passwd"])
76 # submissions = form.getSubmissions(project_id, task, 0, False, True)
77 subs = form.listSubmissions(project_id, task, filters)
78 if not subs:
79 log.error(f"Failed to get submissions for project ({project_id}) task ({task})")
80 continue
81 # log.debug(f"There are {len(subs)} submissions for {task}")
82 if len(subs["value"]) > 0:
83 data += subs["value"]
84 # log.debug(f"There are {len(xforms)} Xforms, and {len(submissions)} submissions total")
85 timer.stop()
86 return data

89class OdkCentral(object):
90 def __init__(
91 self,
92 url: Optional[str] = None,
93 user: Optional[str] = None,
94 passwd: Optional[str] = None,
95 ):
96 """A Class for accessing an ODK Central server via it's REST API.
97
98 Args:
99 url (str): The URL of the ODK Central
100 user (str): The user's account name on ODK Central
101 passwd (str): The user's account password on ODK Central
102
103 Returns:
104 (OdkCentral): An instance of this class
105 """
106 if not url:
107 url = os.getenv("ODK_CENTRAL_URL", default=None)
108 self.url = url
109 if not user:
110 user = os.getenv("ODK_CENTRAL_USER", default=None)
111 self.user = user
112 if not passwd:
113 passwd = os.getenv("ODK_CENTRAL_PASSWD", default=None)
114 self.passwd = passwd
115 verify = os.getenv("ODK_CENTRAL_SECURE", default=True)
116 if type(verify) == str:
117 self.verify = verify.lower() in ("true", "1", "t")
118 else:
119 self.verify = verify
120 # Set cert bundle path for requests in environment
121 if self.verify:
122 os.environ["REQUESTS_CA_BUNDLE"] = "/etc/ssl/certs/ca-certificates.crt"
123 # These are settings used by ODK Collect
124 self.general = {
125 "form_update_mode": "match_exactly",
126 "autosend": "wifi_and_cellular",
127 }
128 # If there is a config file with authentication setting, use that
129 # so we don't have to supply this all the time. This is only used
130 # when odk_client is used, and no parameters are passed in.
131 if not self.url:
132 # log.debug("Configuring ODKCentral from file .odkcentral")
133 home = os.getenv("HOME")
134 config = ".odkcentral"
135 filespec = home + "/" + config
136 if os.path.exists(filespec):
137 file = open(filespec, "r")
138 for line in file:
139 # Support embedded comments
140 if line[0] == "#":
141 continue
142 # Read the config file for authentication settings
143 tmp = line.split("=")
144 if tmp[0] == "url":
145 self.url = tmp[1].strip("\n")
146 if tmp[0] == "user":
147 self.user = tmp[1].strip("\n")
148 if tmp[0] == "passwd":
149 self.passwd = tmp[1].strip("\n")
150 else:
151 log.warning(f"Authentication settings missing from {filespec}")
152 else:
153 log.debug(f"ODKCentral configuration parsed: {self.url}")
154 # Base URL for the REST API
155 self.version = "v1"
156 # log.debug(f"Using {self.version} API")
157 self.base = self.url + "/" + self.version + "/"
158
159 # Use a persistant connect, better for multiple requests
160 self.session = requests.Session()
161
162 # Authentication with session token
163 self.authenticate()
164
165 # These are just cached data from the queries
166 self.projects = dict()
167 self.users = list()

169 def authenticate(
170 self,
171 url: str = None,
172 user: str = None,
173 passwd: str = None,
174 ):
175 """Setup authenticate to an ODK Central server.
176
177 Args:
178 url (str): The URL of the ODK Central
179 user (str): The user's account name on ODK Central
180 passwd (str): The user's account password on ODK Central
181
182 Returns:
183 (requests.Response): A response from ODK Central after auth
184 """
185 if not self.url:
186 self.url = url
187 if not self.user:
188 self.user = user
189 if not self.passwd:
190 self.passwd = passwd
191
192 # Get a session token
193 try:
194 response = self.session.post(
195 f"{self.base}sessions",
196 json={
197 "email": self.user,
198 "password": self.passwd,
199 },
200 )
201 except requests.exceptions.ConnectionError as request_error:
202 # URL does not exist
203 raise ConnectionError("Failed to connect to Central. Is the URL valid?") from request_error
204
205 if response.status_code == 401:
206 # Unauthorized, invalid credentials
207 raise ConnectionError("ODK credentials are invalid, or may have changed. Please update them.") from None
208 elif not response.ok:
209 # Handle other errors
210 response.raise_for_status()
211
212 self.session.headers.update({"Authorization": f"Bearer {response.json().get('token')}"})
213
214 # Connect to the server
215 return self.session.get(self.url, verify=self.verify)

217 def listProjects(self):
218 """Fetch a list of projects from an ODK Central server, and
219 store it as an indexed list.
220
221 Returns:
222 (list): A list of projects on a ODK Central server
223 """
224 log.info("Getting a list of projects from %s" % self.url)
225 url = f"{self.base}projects"
226 result = self.session.get(url, verify=self.verify)
227 projects = result.json()
228 for project in projects:
229 if isinstance(project, dict):
230 if project.get("id") is not None:
231 self.projects[project["id"]] = project
232 else:
233 log.info("No projects returned. Is this a first run?")
234 return projects

236 def createProject(
237 self,
238 name: str,
239 ) -> dict:
240 """Create a new project on an ODK Central server if it doesn't
241 already exist.
242
243 Args:
244 name (str): The name for the new project
245
246 Returns:
247 (json): The response from ODK Central
248 """
249 log.debug(f"Checking if project named {name} exists already")
250 exists = self.findProject(name=name)
251 if exists:
252 log.debug(f"Project named {name} already exists.")
253 return exists
254 else:
255 url = f"{self.base}projects"
256 log.debug(f"POSTing project {name} to {url} with verify={self.verify}")
257 try:
258 result = self.session.post(url, json={"name": name}, verify=self.verify, timeout=4)
259 result.raise_for_status()
260 except requests.exceptions.RequestException as e:
261 log.error(e)
262 log.error("Failed to submit to ODKCentral")
263 json_response = result.json()
264 log.debug(f"Returned: {json_response}")
265 # update the internal list of projects
266 self.listProjects()
267 return json_response

431 def getAllSubmissions(self, project_id: int, xforms: list = None, filters: dict = None):
432 """Fetch a list of submissions in a project on an ODK Central server.
433
434 Args:
435 project_id (int): The ID of the project on ODK Central
436 xforms (list): The list of XForms to get the submissions of
437
438 Returns:
439 (json): All of the submissions for all of the XForm in a project
440 """
441 # The number of threads is based on the CPU cores
442 info = get_cpu_info()
443 self.cores = info["count"]
444
445 timer = Timer(text="getAllSubmissions() took {seconds:.0f}s")
446 timer.start()
447 if not xforms:
448 xforms_data = self.listForms(project_id)
449 xforms = [d["xmlFormId"] for d in xforms_data]
450
451 chunk = round(len(xforms) / self.cores) if round(len(xforms) / self.cores) > 0 else 1
452 last_slice = len(xforms) if len(xforms) % chunk == 0 else len(xforms) - 1
453 cycle = range(0, (last_slice + chunk) + 1, chunk)
454 future = None
455 result = None
456 previous = 0
457 newdata = list()
458
459 # single threaded for easier debugging
460 # for current in cycle:
461 # if previous == current:
462 # continue
463 # result = downloadThread(project_id, xforms[previous:current])
464 # previous = current
465 # newdata += result
466
467 odk_credentials = {"url": self.url, "user": self.user, "passwd": self.passwd}
468
469 with concurrent.futures.ThreadPoolExecutor(max_workers=self.cores) as executor:
470 futures = list()
471 for current in cycle:
472 if previous == current:
473 continue
474 result = executor.submit(downloadThread, project_id, xforms[previous:current], odk_credentials, filters)
475 previous = current
476 futures.append(result)
477 for future in concurrent.futures.as_completed(futures):
478 log.debug("Waiting for thread to complete..")
479 data = future.result(timeout=10)
480 if len(data) > 0:
481 newdata += data
482 timer.stop()
483 return newdata

1048 def createForm(
1049 self,
1050 projectId: int,
1051 data: Union[str, Path, BytesIO],
1052 form_name: Optional[str] = None,
1053 publish: Optional[bool] = False,
1054 ) -> Optional[str]:
1055 """Create a new form on an ODK Central server.
1056
1057 - If no form_name is passed, the form name is generated by default in draft.
1058 If the publish param is also passed, then the form is published.
1059 - If form_name is passed, a new form is created from this in draft state.
1060 This copies across all attachments.
1061
1062 Note:
1063 The form name (xmlFormId) is generated from the id="" attribute
1064 immediately inside the <instance> tag of the XForm XML.
1065
1066 Args:
1067 projectId (int): The ID of the project on ODK Central
1068 form_name (str): The user friendly name to provide the form
1069 data (str, Path, BytesIO): The XForm file path, or BytesIO memory obj
1070 publish (bool): If the new form should be published.
1071 Only valid if form_name is not passed, i.e. a new form.
1072
1073 Returns:
1074 (str, Optional): The form name, else None if failure.
1075 """
1076 # BytesIO memory object
1077 if isinstance(data, BytesIO):
1078 self.xml = data.getvalue().decode("utf-8")
1079 # Filepath
1080 elif isinstance(data, str) or isinstance(data, Path):
1081 xml_path = Path(data)
1082 if not xml_path.exists():
1083 log.error(f"File does not exist on disk: {data}")
1084 return None
1085 # Read the XML or XLS file
1086 with open(xml_path, "rb") as xml_file:
1087 self.xml = xml_file.read()
1088 log.debug("Read %d bytes from %s" % (len(self.xml), data))
1089

1091 self.draft = True
1092 log.debug(f"Creating draft from template form: {form_name}")
1093 url = f"{self.base}projects/{projectId}/forms/{form_name}/draft?ignoreWarnings=true"
1094 else:
1095 # This is not a draft form, its an entirely new form (even if publish=false)
1096 log.debug("Creating new form, with name determined from form_id field")
1097 self.published = True if publish else False
1098 url = f"{self.base}projects/{projectId}/forms?ignoreWarnings=true&{'publish=true' if publish else ''}"
1099
1100 result = self.session.post(
1101 url, data=self.xml, headers=dict({"Content-Type": "application/xml"}, **self.session.headers), verify=self.verify
1102 )
1103
1104 if result.status_code != 200:
1105 try:
1106 status = result.json()
1107 msg = status.get("message", "Unknown error")
1108 if result.status_code == 409:
1109 log.warning(msg)
1110 last_full_stop_index = msg.rfind(".")
1111 last_comma_index = msg.rfind(",")
1112 if last_full_stop_index != -1 and last_comma_index != -1:
1113 # Extract xmlFormId from error msg
1114 xmlFormId = msg[last_comma_index + 1 : last_full_stop_index].strip()
1115 return xmlFormId
1116 else:
1117 log.warning("Unable to extract xmlFormId from error message")
1118 return None
1119 else:
1120 log.error(f"Couldn't create {form_name} on Central: {msg}")
1121 return None
1122 except json.decoder.JSONDecodeError:
1123 log.error(f"Couldn't create {form_name} on Central: Error decoding JSON response")
1124 return None
1125
1126 try:
1127 # Log response to terminal
1128 json_data = result.json()
1129 except json.decoder.JSONDecodeError:
1130 log.error("Could not parse response json during form creation")
1131 return None
1132
1133 # epdb.st()
1134 # FIXME: should update self.forms with the new form
1135
1136 if "success" in json_data:
1137 log.debug(f"Created draft XForm on ODK server: ({form_name})")
1138 return form_name
1139
1140 new_form_name = json_data.get("xmlFormId")
1141 log.info(f"Created XForm on ODK server: ({new_form_name})")
1142 return new_form_name
