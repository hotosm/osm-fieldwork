94def standardize_xlsform_sheets(xlsform: dict) -> dict:
95 """Standardizes column headers in both the 'survey' and 'choices' sheets of an XLSForm.
96
97 - Strips spaces and lowercases all column headers.
98 - Fixes formatting for columns with '::' (e.g., multilingual labels).
99
100 Args:
101 xlsform (dict): A dictionary with keys 'survey' and 'choices', each containing a DataFrame.
102
103 Returns:
104 dict: The updated XLSForm dictionary with standardized column headers.
105 """

107 def standardize_language_columns(df):
108 """Standardize existing language columns.
109
110 :param df: Original DataFrame with existing translations.
111 :param DEFAULT_LANGAUGES: List of DEFAULT_LANGAUGES with their short codes, e.g., {"english": "en", "french": "fr"}.
112 :param base_columns: List of base columns to check (e.g., 'label', 'hint', 'required_message').
113 :return: Updated DataFrame with standardized and complete language columns.
114 """
115 base_columns = ["label", "hint", "required_message"]
116 df.columns = df.columns.str.lower()
117 existing_columns = df.columns.tolist()
118
119 # Map existing columns and standardize their names
120 for col in existing_columns:
121 standardized_col = col
122 for base_col in base_columns:
123 if col.startswith(f"{base_col}::"):
124 match = re.match(rf"{base_col}::\s*(\w+)", col)
125 if match:
126 lang_name = match.group(1)
127 if lang_name in DEFAULT_LANGUAGES:
128 standardized_col = f"{base_col}::{lang_name}({DEFAULT_LANGUAGES[lang_name]})"
129
130 elif col == base_col: # if only label,hint or required_message then add '::english(en)'
131 standardized_col = f"{base_col}::english(en)"
132
133 if col != standardized_col:
134 df.rename(columns={col: standardized_col}, inplace=True)
135 return df

137 def filter_df_empty_rows(df: pd.DataFrame, column: str = NAME_COLUMN):
138 """Remove rows with None values in the specified column.
139
140 NOTE We retain 'end group' and 'end group' rows even if they have no name.
141 NOTE A generic df.dropna(how="all") would not catch accidental spaces etc.
142 """
143 if column in df.columns:
144 # Only retain 'begin group' and 'end group' if 'type' column exists
145 if "type" in df.columns:
146 return df[(df[column].notna()) | (df["type"].isin(["begin group", "end group", "begin_group", "end_group"]))]
147 else:
148 return df[df[column].notna()]
149 return df
150
151 for sheet_name, sheet_df in xlsform.items():
152 if sheet_df.empty:
153 continue
154 # standardize the language columns
155 sheet_df = standardize_language_columns(sheet_df)
156 sheet_df = filter_df_empty_rows(sheet_df)
157 xlsform[sheet_name] = sheet_df
158
159 return xlsform

184def normalize_with_meta(row, meta_df):

193def merge_dataframes(mandatory_df: pd.DataFrame, user_question_df: pd.DataFrame, digitisation_df: pd.DataFrame) -> pd.DataFrame:
194 """Merge multiple Pandas dataframes together, removing duplicate fields."""
195 if "list_name" in user_question_df.columns:
196 merged_df = pd.concat(
197 [
198 mandatory_df,
199 user_question_df,
200 digitisation_df,
201 ],
202 ignore_index=True,
203 )
204 # NOTE here we remove duplicate PAIRS based on `list_name` and the name column
205 # we have `allow_duplicate_choices` set in the settings sheet, so it is possible
206 # to have duplicate NAME_COLUMN entries, if they are in different a `list_name`.
207 return merged_df.drop_duplicates(subset=["list_name", NAME_COLUMN], ignore_index=True)
208
209 user_question_df = user_question_df.apply(normalize_with_meta, axis=1, meta_df=meta_df)
210
211 # Find common fields between user_question_df and mandatory_df or digitisation_df
212 duplicate_fields = set(user_question_df[NAME_COLUMN]).intersection(
213 set(mandatory_df[NAME_COLUMN]).union(set(digitisation_df[NAME_COLUMN]))
214 )
215
216 # NOTE filter out 'end group' from duplicate check as they have empty NAME_COLUMN
217 end_group_rows = user_question_df[user_question_df["type"].isin(["end group", "end_group"])]
218 user_question_df_filtered = user_question_df[
219 (~user_question_df[NAME_COLUMN].isin(duplicate_fields)) | (user_question_df.index.isin(end_group_rows.index))
220 ]
221
222 # Create survey group wrapper
223 survey_group = create_survey_group()
224
225 # Concatenate dataframes in the desired order
226 return pd.concat(
227 [
228 mandatory_df,
229 # Wrap the survey question in a group
230 survey_group["begin"],
231 user_question_df_filtered,
232 survey_group["end"],
233 digitisation_df,
234 ],
235 ignore_index=True,
236 )

239def append_select_one_from_file_row(df: pd.DataFrame, entity_name: str) -> pd.DataFrame:
240 """Add a new select_one_from_file question to reference an Entity."""
241 # Find the row index where name column = 'feature'
242 select_one_from_file_index = df.index[df[NAME_COLUMN] == FEATURE_COLUMN].tolist()
243 if not select_one_from_file_index:
244 raise ValueError(f"Row with '{NAME_COLUMN}' == '{FEATURE_COLUMN}' not found in survey sheet.")
245
246 # Find the row index after 'feature' row
247 row_index_to_split_on = select_one_from_file_index[0] + 1
248
249 additional_row = pd.DataFrame(
250 {
251 "type": [f"select_one_from_file {entity_name}.csv"],
252 "name": [entity_name],
253 "appearance": ["map"],
254 "label::english(en)": [entity_name],
255 "label::swahili(sw)": [entity_name],
256 "label::french(fr)": [entity_name],
257 "label::spanish(es)": [entity_name],
258 }
259 )
260
261 # Prepare the row for calculating coordinates based on the additional entity
262 coordinates_row = pd.DataFrame(
263 {
264 "type": ["calculate"],
265 "name": [f"{entity_name}_geom"],
266 "calculation": [f"instance('{entity_name}')/root/item[name=${{{entity_name}}}]/geometry"],
267 "label::english(en)": [f"{entity_name}_geom"], # translations not needed, calculated field
268 }
269 )
270 # Insert the new row into the DataFrame
271 top_df = df.iloc[:row_index_to_split_on]
272 bottom_df = df.iloc[row_index_to_split_on:]
273 return pd.concat([top_df, additional_row, coordinates_row, bottom_df], ignore_index=True)

276async def append_mandatory_fields(
277 custom_form: BytesIO,
278 form_name: str = f"fmtm_{uuid4()}",
279 additional_entities: list[str] = None,
280 new_geom_type: DbGeomType = DbGeomType.POINT,
281) -> tuple[str, BytesIO]:
282 """Append mandatory fields to the XLSForm for use in FMTM.
283
284 Args:
285 custom_form(BytesIO): the XLSForm data uploaded, wrapped in BytesIO.
286 form_name(str): the friendly form name in ODK web view.
287 additional_entities(list[str]): add extra select_one_from_file fields to
288 reference an additional Entity list (set of geometries).
289 The values should be plural, so that 's' will be stripped in the
290 field name.
291 new_geom_type (DbGeomType): the type of geometry required when collecting
292 new geometry data: point, line, polygon.
293
294 Returns:
295 tuple(str, BytesIO): the xFormId + the update XLSForm wrapped in BytesIO.
296 """
297 log.info("Appending field mapping questions to XLSForm")
298 custom_sheets = pd.read_excel(custom_form, sheet_name=None, engine="calamine")
299
300 if "survey" not in custom_sheets:
301 msg = "Survey sheet is required in XLSForm!"
302 log.error(msg)
303 raise ValueError(msg)
304
305 custom_sheets = standardize_xlsform_sheets(custom_sheets)
306
307 log.debug("Merging survey sheet XLSForm data")
308 survey_df = create_survey_df(new_geom_type)
309 custom_sheets["survey"] = merge_dataframes(survey_df, custom_sheets.get("survey"), digitisation_df)
310
311 # Ensure the 'choices' sheet exists in custom_sheets
312 if "choices" not in custom_sheets or custom_sheets["choices"] is None:
313 custom_sheets["choices"] = pd.DataFrame(columns=["list_name", "name", "label::english(en)"])
314
315 log.debug("Merging choices sheet XLSForm data")
316 custom_sheets["choices"] = merge_dataframes(choices_df, custom_sheets.get("choices"), digitisation_choices_df)
317
318 # Append or overwrite 'entities' and 'settings' sheets
319 log.debug("Overwriting entities and settings XLSForm sheets")
320 custom_sheets["entities"] = entities_df
321 if "entities" not in custom_sheets:
322 msg = "Entities sheet is required in XLSForm!"
323 log.error(msg)
324 raise ValueError(msg)
325 if "settings" not in custom_sheets:
326 msg = "Settings sheet is required in XLSForm!"
327 log.error(msg)
328 raise ValueError(msg)
329
330 # Extract existing form id if present, else set to random uuid
331 if "form_id" in custom_sheets["settings"]:
332 xform_id = custom_sheets["settings"]["form_id"]
333 log.debug(f"Extracted existing form_id field: {xform_id}")
334 else:
335 xform_id = str(uuid4())
336
337 current_datetime = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
338 log.debug(f"Setting xFormId = {xform_id} | version = {current_datetime} | form_name = {form_name}")
339
340 # Set the 'version' column to the current timestamp
341 custom_sheets["settings"]["version"] = current_datetime
342 custom_sheets["settings"]["form_id"] = xform_id
343 custom_sheets["settings"]["form_title"] = form_name
344 if "default_language" not in custom_sheets["settings"]:
345 custom_sheets["settings"]["default_language"] = "en"
346
347 # Append select_one_from_file for additional entities
348 if additional_entities:
349 log.debug("Adding additional entity list reference to XLSForm")
350 for entity_name in additional_entities:
351 custom_sheets["survey"] = append_select_one_from_file_row(custom_sheets["survey"], entity_name)
352
353 # Return spreadsheet wrapped as BytesIO memory object
354 output = BytesIO()
355 with pd.ExcelWriter(output, engine="openpyxl") as writer:
356 for sheet_name, df in custom_sheets.items():
357 df.to_excel(writer, sheet_name=sheet_name, index=False)
358 output.seek(0)
359 return (xform_id, output)

362async def main():
363 """Used for the `fmtm_xlsform` CLI command."""
364 parser = argparse.ArgumentParser(description="Append field mapping fields to XLSForm")
365 parser.add_argument("-v", "--verbose", action="store_true", help="verbose output")
366 parser.add_argument("-i", "--input", help="Input XLSForm file")
367 parser.add_argument("-c", "--category", help="A category of demo form to use instead")
368 parser.add_argument("-o", "--output", help="Output merged XLSForm filename")
369 parser.add_argument("-a", "--additional-dataset-names", help="Names of additional entity lists to append")
370 parser.add_argument(
371 "-n",
372 "--new-geom-type",
373 type=DbGeomType,
374 choices=list(DbGeomType),
375 help="The type of new geometry",
376 default=DbGeomType.POINT,
377 )
378 args = parser.parse_args()
379
380 # If verbose, dump to the terminal
381 if args.verbose is not None:
382 logging.basicConfig(
383 level=logging.DEBUG,
384 format=("%(threadName)10s - %(name)s - %(levelname)s - %(message)s"),
385 datefmt="%y-%m-%d %H:%M:%S",
386 stream=sys.stdout,
387 )
388
389 if not args.output:
390 log.error("You must provide an output file with the '-o' flag")
391 parser.print_help()
392
393 if args.input:
394 input_file = Path(args.input)
395 elif args.category:
396 input_file = Path(f"{xlsforms_path}/{args.category}.yaml")
397 else:
398 log.error("Must choose one of '-i' for file input, or '-c' for category selection")
399 parser.print_help()
400 sys.exit(1)
401
402 if not input_file.exists():
403 log.error(f"The file does not exist: {str(input_file)}")
404 sys.exit(1)
405
406 with open(input_file, "rb") as file_handle:
407 input_xlsform = BytesIO(file_handle.read())
408
409 form_id, form_bytes = await append_mandatory_fields(
410 custom_form=input_xlsform,
411 form_name=f"fmtm_{uuid4()}",
412 additional_entities=args.additional_dataset_names,
413 new_geom_type=args.new_geom_type,
414 )
415
416 log.info(f"Form ({form_id}) created successfully")
417 with open(args.output, "wb") as file_handle:
418 file_handle.write(form_bytes.getvalue())
