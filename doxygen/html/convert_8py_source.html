178 def convertEntry(
179 self,
180 tag: str,
181 value: str,
182 ) -> list:
183 """Convert a tag and value from the ODK represention to an OSM one.
184
185 Args:
186 tag (str): The tag from the ODK XML file
187 value (str): The value from the ODK XML file
188
189 Returns:
190 (list): The converted values
191 """
192 all = list()
193
194 # If it's not in any conversion data, pass it through unchanged.
195 if tag.lower() in self.ignore:
196 # logging.debug(f"FIXME: Ignoring {tag}")
197 return None
198 low = tag.lower()
199 if value is None:
200 return low
201
202 if low not in self.convert and low not in self.ignore and low not in self.private:
203 return {tag: value}
204
205 newtag = tag.lower()
206 newval = value
207 # If the tag is in the config file, convert it.
208 if self.convertDataconvertData(newtag):
209 newtag = self.convertTag(newtag)
210 # if newtag != tag:
211 # logging.debug(f"Converted Tag for entry {tag} to {newtag}")
212
213 # Truncate the elevation, as it's really long
214 if newtag == "ele":
215 value = value[:7]
216 newval = self.convertValue(newtag, value)
217 # logging.debug("Converted Value for entry '%s' to '%s'" % (value, newval))
218 # there can be multiple new tag/value pairs for some values from ODK
219 if type(newval) == str:
220 all.append({newtag: newval})
221 elif type(newval) == list:
222 for entry in newval:
223 if type(entry) == str:
224 all.append({newtag: newval})
225 elif type(entry) == dict:
226 for k, v in entry.items():
227 all.append({k: v})
228 return all

230 def convertValue(
231 self,
232 tag: str,
233 value: str,
234 ) -> list:
235 """Convert a single tag value.
236
237 Args:
238 tag (str): The tag from the ODK XML file
239 value (str): The value from the ODK XML file
240
241 Returns:
242 (list): The converted values
243 """
244 all = list()
245
246 vals = self.getValues(tag)
247 # There is no conversion data for this tag
248 if vals is None:
249 return value
250
251 if type(vals) is dict:
252 if value not in vals:
253 all.append({tag: value})
254 return all
255 if type(vals[value]) is bool:
256 entry = dict()
257 if vals[value]:
258 entry[tag] = "yes"
259 else:
260 entry[tag] = "no"
261 all.append(entry)
262 return all
263 for item in vals[value].split(","):
264 entry = dict()
265 tmp = item.split("=")
266 if len(tmp) == 1:
267 entry[tag] = vals[value]
268 else:
269 entry[tmp[0]] = tmp[1]
270 logging.debug("\tValue %s converted value to %s" % (value, entry))
271 all.append(entry)
272 return all

306 def convertMultiple(
307 self,
308 value: str,
309 ) -> list:
310 """Convert a multiple tags from a select_multiple question..
311
312 Args:
313 value (str): The tags from the ODK XML file
314
315 Returns:
316 (list): The new tags
317 """
318 tags = dict()
319 for tag in value.split(" "):
320 low = tag.lower()
321 if self.convertDataconvertData(low):
322 newtag = self.convert[low]
323 if newtag.find("=") > 0:
324 tmp = newtag.split("=")
325 if tmp[0] in tags:
326 tags[tmp[0]] = f"{tags[tmp[0]]};{tmp[1]}"
327 else:
328 tags.update({tmp[0]: tmp[1]})
329 else:
330 tags.update({low: "yes"})
331 # logging.debug(f"\tConverted multiple to {tags}")
332 return tags

334 def parseXLS(
335 self,
336 xlsfile: str,
337 ):
338 """Parse the source XLSFile if available to look for details we need."""
339 if xlsfile is not None and len(xlsfile) > 0:
340 self.entries = pd.read_excel(xlsfile, sheet_name=[0])[0]
341 # There will only be a single sheet
342 names = self.entries["name"]
343 defaults = self.entries["default"]
344 i = 0
345 while i < len(self.entries):
346 if type(self.entries["type"][i]) == float:
347 self.types[self.entries["name"][i]] = None
348 else:
349 self.types[self.entries["name"][i]] = self.entries["type"][i].split(" ")[0]
350 i += 1
351 total = len(names)
352 i = 0
353 while i < total:
354 entry = defaults[i]
355 if str(entry) != "nan":
356 pat = re.compile("..last-saved.*")
357 if pat.match(entry):
358 name = entry.split("#")[1][:-1]
359 self.saved[name] = None
360 else:
361 self.defaults[names[i]] = entry
362 i += 1
363 return True

365 def createEntry(
366 self,
367 entry: dict,
368 ) -> dict:
369 """Create the feature data structure.
370
371 Args:
372 entry (dict): The feature data
373
374 Returns:
375 (dict): The OSM data structure for this entry from the json file
376 """
377 # print(line)
378 feature = dict()
379 attrs = dict()
380 tags = dict()
381 priv = dict()
382 refs = list()
383
384 # log.debug("Creating entry")
385 # First convert the tag to the approved OSM equivalent
386 if "lat" in entry and "lon" in entry:
387 attrs["lat"] = entry["lat"]
388 attrs["lon"] = entry["lon"]
389 for key, value in entry.items():
390 attributes = (
391 "id",
392 "timestamp",
393 "lat",
394 "lon",
395 "uid",
396 "user",
397 "version",
398 "action",
399 )
400
401 if key in self.ignore:
402 continue
403 # When using existing OSM data, there's a special geometry field.
404 # Otherwise use the GPS coordinates where you are.
405 if key == "geometry" and len(value) > 0:
406 geometry = value.split(" ")
407 if len(geometry) == 4:
408 attrs["lat"] = geometry[0]
409 attrs["lon"] = geometry[1]
410 continue
411
412 # if 'lat' in attrs and len(attrs["lat"]) == 0:
413 # continue
414
415 if key is not None and len(key) > 0 and key in attributes:
416 attrs[key] = value
417 # log.debug("Adding attribute %s with value %s" % (key, value))
418 continue
419 if value is not None and value != "no" and value != "unknown":
420 if key == "username":
421 tags["user"] = value
422 continue
423 items = self.convertEntry(key, value)
424 if key in self.types:
425 if self.types[key] == "select_multiple":
426 vals = self.convertMultiple(value)
427 if len(vals) > 0:
428 for tag in vals:
429 tags.update(tag)
430 continue
431 if key == "track" or key == "geoline":
432 # refs.append(tags)
433 # log.debug("Adding reference %s" % tags)
434 refs = value.split(";")
435 elif type(value) != str:
436 if self.privateDataprivateData(key):
437 priv[key] = str(value)
438 else:
439 tags[key] = str(value)
440 elif len(value) > 0:
441 if self.privateDataprivateData(key):
442 priv[key] = value
443 else:
444 tags[key] = value
445 feature["attrs"] = attrs
446 if len(tags) > 0:
447 # logging.debug(f"TAGS: {tags}")
448 feature["tags"] = tags
449 if len(refs) > 1:
450 feature["refs"] = refs
451 if len(priv) > 0:
452 feature["private"] = priv
453
454 return feature
