42def getChoices():
43 """Get the categories and associated XLSFiles from the config file.
44
45 Returns:
46 (list): A list of the XLSForms included in osm-fieldwork
47 """
48 data = dict()
49 if os.path.exists(f"{data_models_path}/category.yaml"):
50 file = open(f"{data_models_path}/category.yaml", "r").read()
51 contents = yaml.load(file, Loader=yaml.Loader)
52 for entry in contents:
53 [[k, v]] = entry.items()
54 data[k] = v[0]
55 return data

79 # Read in the XLSFile
80 if "/" in xlsfile:
81 file = open(xlsfile, "rb")

85 self.config = QueryConfig(config)

87 def getFeatures(
88 self,
89 boundary: FeatureCollection,
90 polygon: bool,
91 ):
92 """Extract features from Postgres.
93
94 Args:
95 boundary (str): The filespec for the project AOI in GeoJson format
96 filespec (str): The optional output file for the query
97 polygon (bool): Whether to have the full geometry or just centroids returns
98
99 Returns:
100 (FeatureCollection): The features returned from the query
101 """
102 log.info("Extracting features from Postgres...")
103
104 if "features" in boundary:
105 poly = boundary["features"][0]["geometry"]
106 else:
107 poly = boundary["geometry"]
108 shape(poly)
109
110 collection = self.db.execQuery(boundary, None, False)
111 if not collection:
112 return None
113
114 return collection

138def main():
139 """This program makes data extracts from OSM data, which can be used with ODK Collect."""
140 getChoices()
141
142 parser = argparse.ArgumentParser(
143 prog="make_data_extract",
144 formatter_class=argparse.RawDescriptionHelpFormatter,
145 description="Make GeoJson data file for ODK from OSM",
146 epilog="""
147 This program makes data extracts for ODK Collect. It uses
148 either a local postgres database, or the remote Underpass
149 database, which is updated every minute for the planet.
150
151 Because ODK Collect won't launch if there are tags in the
152 data extract not also in the choices sheet, this program
153 filters the data based on the XLSForm.
154
155 example:
156 make_data_extract -b [POLY] -c [CONFIG] -x [XLSFORM] -u [URI]
157
158 Where POLY is a boundary polygon for the extract, and
159 CONFIG is YAML or JSON config for the query, and
160 XLSFILE is the XLSForm for this data collection,
161 and URI is the database URI.
162
163 The defaults are buildings.yaml for the config file,
164 buildings.xls for the XLSForm, and for the database,
165 it uses the remote Underpass database.
166 """,
167 )
168 parser.add_argument("-v", "--verbose", nargs="?", const="0", help="verbose output")
169 parser.add_argument("-p", "--polygon", action="store_true", default=False, help="Output polygons instead of centroids")
170 parser.add_argument("-g", "--geojson", default="extract.geojson", help="Name of the GeoJson output file")
171 parser.add_argument("-u", "--uri", default="underpass", help="Database URI")
172 parser.add_argument("-b", "--boundary", required=True, help="Boundary polygon to limit the data size")
173 parser.add_argument("-c", "--config", default="buildings.yaml", help="Config file for the query")
174 parser.add_argument("-x", "--xlsfile", default="buildings.xls", help="XLSForm for this data collection")
175 parser.add_argument("-l", "--list", action="store_true", help="List files in the XLSForm library")
176 args = parser.parse_args()
177
178 if args.list:
179 files = getChoices()
180 for k, v in files.items():
181 print(f"Category: {k}, XLSForm: {v}")
182 quit()
183
184 # if verbose, dump to the terminal.
185 if args.verbose is not None:
186 logging.basicConfig(
187 level=logging.DEBUG,
188 format=("%(asctime)s.%(msecs)03d [%(levelname)s] %(name)s | %(funcName)s:%(lineno)d | %(message)s"),
189 datefmt="%y-%m-%d %H:%M:%S",
190 stream=sys.stdout,
191 )
192
193 extract = MakeExtract(args.uri, args.config, args.xlsfile)
194 file = open(args.boundary, "r")
195 poly = geojson.load(file)
196 data = extract.getFeatures(poly, args.polygon)
197 log.debug(f"Query returned {len(data['features'])} features")
198 # FIXME: just for debugging before filtering!
199 # if len(data['features']) > 0:
200 # jsonfile = open(args.geojson, "w")
201 # dump(data, jsonfile)
202 # jsonfile.close()
203
204 cleaned = extract.cleanFeatures(data)
205
206 jsonfile = open(args.geojson, "w")
207 dump(cleaned, jsonfile)
208 jsonfile.close()
209
210 log.info("Wrote output data file to: %s" % args.geojson)
